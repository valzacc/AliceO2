48 template <typename T, typename ARRAY>
49 class ARROW_EXPORT SortedGroupByKernel : public arrow::compute::UnaryKernel
50 {
51  public:
52  explicit SortedGroupByKernel(GroupByOptions<T> options = {}) : mOptions(options){};
53  arrow::Status Call(arrow::compute::FunctionContext*,
54  arrow::compute::Datum const& table,
55  arrow::compute::Datum* outputRanges) override
56  {
57  using namespace arrow;
58  if (table.kind() == arrow::compute::Datum::TABLE) {
59  auto atable = util::get<std::shared_ptr<arrow::Table>>(table.value);
60  auto columnIndex = atable->schema()->GetFieldIndex(mOptions.columnName);
61  auto chunkedArray = atable->column(columnIndex);
62  return doGrouping(chunkedArray, outputRanges);
63  };
64  return arrow::Status::OK();
65  };

67 #ifdef __clang__
68 #pragma GCC diagnostic ignored "-Winconsistent-missing-override"
69 #endif // __clang__
70  std::shared_ptr<arrow::DataType> out_type() const final
71  {
72  return mType;
73  }
74 #pragma GCC diagnostic pop

76  private:
77  arrow::Status doGrouping(std::shared_ptr<arrow::ChunkedArray> chunkedArray, arrow::compute::Datum* outputRanges)
78  {
79  o2::framework::TableBuilder builder;
80  auto writer = builder.persist<T, T, T>({"start", "count", "index"});
81  auto zeroChunk = std::static_pointer_cast<ARRAY>(chunkedArray->chunk(0));
82  if (zeroChunk->length() == 0) {
83  *outputRanges = std::move(builder.finalize());
84  return arrow::Status::OK();
85  }
86  T currentIndex = 0;
87  T currentCount = 0;
88  T currentOffset = 0;
89  for (auto ci = 0; ci < chunkedArray->num_chunks(); ++ci) {
90  auto chunk = chunkedArray->chunk(ci);
91  T const* data = std::static_pointer_cast<ARRAY>(chunk)->raw_values();
92  for (auto ai = 0; ai < chunk->length(); ++ai) {
93  if (currentIndex == data[ai]) {
94  currentCount++;
95  } else {
96  writer(0, currentOffset, currentCount, currentIndex);
97  currentOffset += currentCount;
98  while (data[ai] - currentIndex > 1) {
99  writer(0, currentOffset, 0, ++currentIndex);
100  }
101  currentIndex++;
102  currentCount = 1;
103  }
104  }
105  }
106  writer(0, currentOffset, currentCount, currentIndex);
107  while (currentIndex < mOptions.size - 1) {
108  writer(0, currentOffset, 0, ++currentIndex);
109  }
110  *outputRanges = std::move(builder.finalize());
111  return arrow::Status::OK();
112  }
113  std::shared_ptr<arrow::DataType> mType;
114  GroupByOptions<T> mOptions;
115 };

170 template <typename T>
176 auto sliceByColumn(char const* key,
177  std::shared_ptr<arrow::Table> const& input,
178  T fullSize,
179  std::vector<arrow::Datum>* slices,
180  std::vector<uint64_t>* offsets = nullptr)
181 {
182  arrow::Datum value_counts;
183  auto options = arrow::compute::CountOptions::Defaults();
184  ARROW_ASSIGN_OR_RAISE(value_counts,
185  arrow::compute::CallFunction("value_counts", {input->GetColumnByName(key)},
186  &options));
187  auto pair = static_cast<arrow::StructArray>(value_counts.array());
188  auto values = static_cast<arrow::NumericArray<typename detail::ConversionTraits<T>::ArrowType>>(pair.field(0)->data());
189  auto counts = static_cast<arrow::NumericArray<arrow::Int64Type>>(pair.field(1)->data());
190 
191  // create slices and offsets
192  auto offset = 0;
193  auto count = 0;
194 
195  auto size = values.length();
196  std::shared_ptr<arrow::Schema> schema(input->schema());
197  std::vector<std::shared_ptr<arrow::ChunkedArray>> sliceArray;
198 
199  auto injectSlice = [&](T count) {
200  for (auto ci = 0; ci < schema->num_fields(); ++ci) {
201  sliceArray.emplace_back(input->column(ci)->Slice(offset, count));
202  }
203  slices->emplace_back(arrow::Datum(arrow::Table::Make(schema, sliceArray)));
204  if (offsets) {
205  offsets->emplace_back(offset);
206  }
207  sliceArray.clear();
208  };
209 
210  auto current = 0;
211  auto v = values.Value(0);
212  while (v - current >= 1) {
213  injectSlice(0);
214  ++current;
215  }
216 
217  for (auto r = 0; r < size - 1; ++r) {
218  count = counts.Value(r);
219  injectSlice(count);
220  offset += count;
221  auto nextValue = values.Value(r + 1);
222  auto value = values.Value(r);
223  while (nextValue - value > 1) {
224  injectSlice(0);
225  ++value;
226  }
227  }
228  injectSlice(counts.Value(size - 1));
229 
230  if (values.Value(size - 1) < fullSize - 1) {
231  for (auto v = values.Value(size - 1) + 1; v < fullSize; ++v) {
232  injectSlice(0);
233  }
234  }
235 
236  return arrow::Status::OK();
237 }
