21 double invmass2prongs2(double px0, double py0, double pz0, double mass0,
22  double px1, double py1, double pz1, double mass1)
23 {
24 
25  double energy0_ = energy(px0, py0, pz0, mass0);
26  double energy1_ = energy(px1, py1, pz1, mass1);
27  double energytot = energy0_ + energy1_;
28 
29  double psum2 = (px0 + px1) * (px0 + px1) +
30  (py0 + py1) * (py0 + py1) +
31  (pz0 + pz1) * (pz0 + pz1);
32  return energytot * energytot - psum2;
33 }
