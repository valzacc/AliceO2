21 bool RunToTimestamp::insert(unsigned int runNumber, long timestamp)
22 {
23  std::pair<std::map<unsigned int, long>::iterator, bool> check;
24  check = mMap.insert(std::pair<unsigned int, long>(runNumber, timestamp));
25  if (!check.second) {
26  LOG(FATAL) << "Run number " << runNumber << " already existed with a timestamp of " << check.first->second;
27  return false;
28  }
29  LOG(INFO) << "Add new run " << runNumber << " with timestamp " << timestamp << " to converter";
30  return true;
31 }

33 bool RunToTimestamp::update(unsigned int runNumber, long timestamp)
34 {
35  if (!Has(runNumber)) {
36  LOG(FATAL) << "Run to Timestamp converter does not have run " << runNumber << ", cannot update converter";
37  return false;
38  }
39  mMap[runNumber] = timestamp;
40  return true;
41 }

43 long RunToTimestamp::getTimestamp(unsigned int runNumber) const
44 {
45  if (!Has(runNumber)) {
46  LOG(ERROR) << "Run to Timestamp converter does not have run " << runNumber;
47  return 0;
48  }
