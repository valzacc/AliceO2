124 template <typename T>
125 std::unique_ptr<TH1> createTH1FromSpec(HistogramSpec const& spec)
126 {
127  if (spec.config.axes.size() == 0) {
128  throw std::runtime_error("No arguments available in spec to create a histogram");
129  }
130  if (spec.config.binsEqual) {
131  return std::make_unique<T>(spec.name.data(), spec.readableName.data(), spec.config.axes[0].nBins, spec.config.axes[0].bins[0], spec.config.axes[0].bins[1]);
132  }
133  return std::make_unique<T>(spec.name.data(), spec.readableName.data(), spec.config.axes[0].nBins, spec.config.axes[0].bins.data());

230 class HistogramRegistry
231 {
232  public:
233  HistogramRegistry(char const* const name_, bool enable, std::vector<HistogramSpec> specs)
234  : name(name_),
235  enabled(enable),
236  mRegistryKey(),
237  mRegistryValue(),
238  mHistogramCreationCallbacks({HistogramCallbacks::createTH1D(),
239  HistogramCallbacks::createTH1F(),
240  HistogramCallbacks::createTH1I(),
241  HistogramCallbacks::createTH2D(),
242  HistogramCallbacks::createTH2F(),
243  HistogramCallbacks::createTH2I(),
244  HistogramCallbacks::createTH3D(),
245  HistogramCallbacks::createTH3F(),
246  HistogramCallbacks::createTH3I()})
247  {
248  mRegistryKey.fill(0u);
249  for (auto& spec : specs) {
250  insert(spec);
251  }
252  }
253 
255  auto& get(char const* const name) const
256  {
257  const uint32_t id = compile_time_hash(name);
258  const uint32_t i = imask(id);
259  if (O2_BUILTIN_LIKELY(id == mRegistryKey[i])) {
260  return mRegistryValue[i];
261  }
262  for (auto j = 1u; j < MAX_REGISTRY_SIZE; ++j) {
263  if (id == mRegistryKey[imask(j + i)]) {
264  return mRegistryValue[imask(j + i)];
265  }
266  }
267  throw std::runtime_error("No match found!");
268  }
269 
271  auto& operator()(char const* const name) const
272  {
273  return get(name);
274  }
275 
276  // @return the associated OutputSpec
277  OutputSpec const spec()
278  {
279  ConcreteDataMatcher matcher{"HIST", "\0", 0};
280  strncpy(matcher.description.str, this->name.data(), 16);
281  return OutputSpec{OutputLabel{this->name}, matcher};
282  }
283 
284  OutputRef ref()
285  {
286  return OutputRef{this->name, 0};
287  }
288 
290  mutable uint32_t lookup = 0;
291 
292  private:
293  void insert(HistogramSpec& spec)
294  {
295  uint32_t i = imask(spec.id);
296  for (auto j = 0u; j < MAX_REGISTRY_SIZE; ++j) {
297  if (mRegistryValue[imask(j + i)].get() == nullptr) {
298  mRegistryKey[imask(j + i)] = spec.id;
299  mRegistryValue[imask(j + i)] = mHistogramCreationCallbacks[spec.config.type](spec);
300  lookup += j;
301  return;
302  }
303  }
304  throw std::runtime_error("Internal array is full.");
305  }
306 
307  inline constexpr uint32_t imask(uint32_t i) const
308  {
309  return i & mask;
310  }
311  std::string name;
312  bool enabled;
313 
316  static constexpr uint32_t mask = 0x1FF;
317  static constexpr uint32_t MAX_REGISTRY_SIZE = mask + 1;
318  std::array<uint32_t, MAX_REGISTRY_SIZE> mRegistryKey;
319  std::array<std::unique_ptr<TH1>, MAX_REGISTRY_SIZE> mRegistryValue;
320  std::vector<HistogramCreationCallback> mHistogramCreationCallbacks;
321 };
