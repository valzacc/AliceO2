40  void process(aod::HfCandProng2 const& candidates)
41  {
42  for (auto& candidate : candidates) {
43  hmass->Fill(InvMassD0(candidate));
44  hmass->Fill(InvMassD0bar(candidate));
45  hptcand->Fill(candidate.pt());
46  hptprong0->Fill(candidate.ptProng0());
47  hptprong1->Fill(candidate.ptProng1());
48  hdeclength->Fill(candidate.decayLength());
49  hdeclengthxy->Fill(candidate.decayLengthXY());
50  hd0->Fill(candidate.impactParameter0());
51  hd0->Fill(candidate.impactParameter1());
52  hd0d0->Fill(candidate.impactParameterProduct());
53  hCTS->Fill(CosThetaStarD0(candidate));
54  hCt->Fill(CtD0(candidate));
55  hEta->Fill(candidate.eta());
56  }
57  }
58 };
