106  void process(soa::Join<aod::ReducedEvents, aod::ReducedEventsExtended, aod::ReducedEventsVtxCov>::iterator event,
107  soa::Join<aod::ReducedTracks, aod::ReducedTracksBarrel, aod::ReducedTracksBarrelCov, aod::ReducedTracksBarrelPID> const& tracks,
108  soa::Join<aod::ReducedMuons, aod::ReducedMuonsExtended> const& muons)
109  {
110  // Reset the fgValues array
111  // TODO: reseting will have to be done selectively, for example run-wise variables don't need to be reset every event, but just updated if the run changes
112  // The reset can be done selectively, using arguments in the ResetValues() function
113  VarManager::ResetValues();
114 
115  VarManager::FillEvent<fgEventFillMap>(event);
116  fHistMan->FillHistClass("Event_BeforeCuts", VarManager::fgValues); // automatically fill all the histograms in the class Event
117  if (!fEventCut->IsSelected(VarManager::fgValues))
118  return;
119  fHistMan->FillHistClass("Event_AfterCuts", VarManager::fgValues);
120 
121  // loop over barrel tracks and store positive and negative tracks in separate arrays
122  // TODO: use Partition initiaslized by vector of track indices when this will be available
123  std::vector<std::decay_t<decltype(tracks)>::iterator> selectedTracksPos;
124  std::vector<std::decay_t<decltype(tracks)>::iterator> selectedTracksNeg;
125  for (auto& track : tracks) {
126 
127  VarManager::ResetValues(VarManager::kNEventWiseVariables, VarManager::kNMuonTrackVariables);
128 
129  VarManager::FillTrack<fgTrackFillMap>(track);
130  fHistMan->FillHistClass("TrackBarrel_BeforeCuts", VarManager::fgValues);
131  if (!fTrackCut->IsSelected(VarManager::fgValues))
132  continue;
133  fHistMan->FillHistClass("TrackBarrel_AfterCuts", VarManager::fgValues);
134 
135  if (track.charge() < 0)
136  selectedTracksNeg.push_back(track);
137  if (track.charge() > 0)
138  selectedTracksPos.push_back(track);
139  }
140 
141  // run the same event pairing for barrel tracks
142  for (auto& tpos : selectedTracksPos) {
143  for (auto& tneg : selectedTracksNeg) {
144  VarManager::FillPair(tpos, tneg);
145  fHistMan->FillHistClass("PairsBarrel", VarManager::fgValues);
146  }
147  }
148 
149  // loop over muon tracks
150  std::vector<std::decay_t<decltype(muons)>::iterator> selectedMuonsPos;
151  std::vector<std::decay_t<decltype(muons)>::iterator> selectedMuonsNeg;
152  for (auto& muon : muons) {
153  VarManager::FillTrack<fgMuonFillMap>(muon);
154  fHistMan->FillHistClass("TrackMuon_BeforeCuts", VarManager::fgValues);
155  if (!fMuonCut->IsSelected(VarManager::fgValues))
156  continue;
157  fHistMan->FillHistClass("TrackMuon_AfterCuts", VarManager::fgValues);
158 
159  if (muon.charge() < 0)
160  selectedMuonsNeg.push_back(muon);
161  if (muon.charge() > 0)
162  selectedMuonsPos.push_back(muon);
163  }
164 
165  // same event pairing for muons
166  for (auto& tpos : selectedMuonsNeg) {
167  for (auto& tneg : selectedMuonsPos) {
168  VarManager::FillPair(tpos, tneg);
169  fHistMan->FillHistClass("PairsMuon", VarManager::fgValues);
170  }
171  }
172  }
