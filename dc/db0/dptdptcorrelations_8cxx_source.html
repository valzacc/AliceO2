117 bool IsEvtSelected(aod::CollisionEvSelCent const& collision)
118 {
119  if (collision.alias()[kINT7]) {
120  if (collision.sel7()) {
121  if (zvtxlow < collision.posZ() and collision.posZ() < zvtxup) {
122  return true;
123  }
124  return false;
125  }
126  }
127  return false;
128 }

130 bool matchTrackType(aod::TrackData const& track)
131 {
132  switch (cfgTrackType) {
133  case 1:
134  if (track.isGlobalTrack() != 0)
135  return true;
136  else
137  return false;
138  break;
139  case 2:
140  if (track.isGlobalTrackSDD() != 0)
141  return true;
142  else
143  return false;
144  break;
145  default:
146  return false;
147  }
148 }

150 std::tuple<uint8_t, uint8_t> AcceptTrack(aod::TrackData const& track)
151 {
152 
153  uint8_t asone = DPTDPT_FALSE;
154  uint8_t astwo = DPTDPT_FALSE;
155 
156  /* TODO: incorporate a mask in the scanned tracks table for the rejecting track reason */
157  if (matchTrackType(track)) {
158  if (ptlow < track.pt() and track.pt() < ptup and etalow < track.eta() and track.eta() < etaup) {
159  if (((track.charge() > 0) and (cfgTrackOneCharge > 0)) or ((track.charge() < 0) and (cfgTrackOneCharge < 0))) {
160  asone = DPTDPT_TRUE;
161  } else if (((track.charge() > 0) and (cfgTrackTwoCharge > 0)) or ((track.charge() < 0) and (cfgTrackTwoCharge < 0))) {
162  astwo = DPTDPT_TRUE;
163  }
164  }
165  }
166  return std::make_tuple(asone, astwo);
167 }

256  void process(aod::CollisionEvSelCent const& collision, soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra, aod::TracksExtended, aod::TrackSelection> const& ftracks)
257  {
258  // LOGF(INFO,"New collision with %d filtered tracks", ftracks.size());
259  fhCentMultB->Fill(collision.centV0M());
260  fhVertexZB->Fill(collision.posZ());
261  int acceptedevent = DPTDPT_FALSE;
262  if (IsEvtSelected(collision)) {
263  acceptedevent = DPTDPT_TRUE;
264  fhCentMultA->Fill(collision.centV0M());
265  fhVertexZA->Fill(collision.posZ());
266  // LOGF(INFO,"New accepted collision with %d filtered tracks", ftracks.size());
267 
268  for (auto& track : ftracks) {
269  /* before track selection */
270  fhPtB->Fill(track.pt());
271  fhEtaB->Fill(track.eta());
272  fhPhiB->Fill(track.phi());
273  // no more histograms for the time being
274  // fhEtaVsPhiB->Fill(track.phi(),track.eta());
275  // fhPtVsEtaB->Fill(track.eta(),track.pt());
276  if (track.charge() > 0) {
277  fhPtPosB->Fill(track.pt());
278  } else {
279  fhPtNegB->Fill(track.pt());
280  }
281 
282  /* track selection */
283  /* tricky because the boolean columns issue */
284  auto [asone, astwo] = AcceptTrack(track);
285  if ((asone == DPTDPT_TRUE) or (astwo == DPTDPT_TRUE)) {
286  /* the track has been accepted */
287  fhPtA->Fill(track.pt());
288  fhEtaA->Fill(track.eta());
289  fhPhiA->Fill(track.phi());
290  // no more histograms for the time being
291  // fhEtaVsPhiA->Fill(track.phi(),track.eta());
292  // fhPtVsEtaA->Fill(track.eta(),track.pt());
293  if (track.charge() > 0) {
294  fhPtPosA->Fill(track.pt());
295  } else {
296  fhPtNegA->Fill(track.pt());
297  }
298  }
299  scannedtracks(asone, astwo);
300  }
301  } else {
302  for (auto& track : ftracks) {
303  scannedtracks(DPTDPT_FALSE, DPTDPT_FALSE);
304  }
305  }
306  acceptedevents(acceptedevent);
307  }
308 };

344 
345  OutputObj<TH1F> fTracksOne{TH1F("TracksOne", "Tracks as track one;number of tracks;events", 1500, 0.0, 1500.0)};
346  OutputObj<TH1F> fTracksTwo{TH1F("TracksTwo", "Tracks as track two;number of tracks;events", 1500, 0.0, 1500.0)};

350  OutputObj<TH1F> fTracksTwoUnsel{TH1F("TracksTwoUnsel", "Tracks as track two;number of tracks;events", 1500, 0.0, 1500.0)};
351  OutputObj<TH1F> fTracksOneAndTwoUnsel{TH1F("TracksOneAndTwoUnsel", "Tracks as track one and as track two;number of tracks;events", 1500, 0.0, 1500.0)};
352  OutputObj<TH1F> fTracksNoneUnsel{TH1F("TracksNoneUnsel", "Not selected tracks;number of tracks;events", 1500, 0.0, 1500.0)};
353  OutputObj<TH1F> fSelectedEvents{TH1F("SelectedEvents", "Selected events;number of tracks;events", 2, 0.0, 2.0)};
354 
355  Filter onlyacceptedevents = (aod::dptdptcorrelations::eventaccepted == DPTDPT_TRUE);
356  Filter onlyacceptedtracks = ((aod::dptdptcorrelations::trackacceptedasone == DPTDPT_TRUE) or (aod::dptdptcorrelations::trackacceptedastwo == DPTDPT_TRUE));
357 
358  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels, aod::Cents, aod::AcceptedEvents>>::iterator const& collision,
359  soa::Filtered<soa::Join<aod::Tracks, aod::ScannedTracks>> const& tracks)
360  {
361  if (collision.eventaccepted() != DPTDPT_TRUE) {
362  fSelectedEvents->Fill(0.5);
363  } else {
364  fSelectedEvents->Fill(1.5);
365  }
366 
367  int ntracks_one = 0;
368  int ntracks_two = 0;
369  int ntracks_one_and_two = 0;
370  int ntracks_none = 0;
371  for (auto& track : tracks) {
372  if ((track.trackacceptedasone() != DPTDPT_TRUE) and (track.trackacceptedastwo() != DPTDPT_TRUE)) {
373  ntracks_none++;
374  }
375  if ((track.trackacceptedasone() == DPTDPT_TRUE) and (track.trackacceptedastwo() == DPTDPT_TRUE)) {
376  ntracks_one_and_two++;
377  }
378  if (track.trackacceptedasone() == DPTDPT_TRUE) {
379  ntracks_one++;
380  }
381  if (track.trackacceptedastwo() == DPTDPT_TRUE) {
382  ntracks_two++;
383  }
384  }
385  if (collision.eventaccepted() != DPTDPT_TRUE) {
386  /* control for non selected events */
387  fTracksOneUnsel->Fill(ntracks_one);
388  fTracksTwoUnsel->Fill(ntracks_two);
389  fTracksNoneUnsel->Fill(ntracks_none);
390  fTracksOneAndTwoUnsel->Fill(ntracks_one_and_two);
391  } else {
392  fTracksOne->Fill(ntracks_one);
393  fTracksTwo->Fill(ntracks_two);
394  fTracksNone->Fill(ntracks_none);
395  fTracksOneAndTwo->Fill(ntracks_one_and_two);
396  }
397  }
