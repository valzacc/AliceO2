213 struct IndexExclusive {
215  template <typename... Cs, typename Key, typename T1, typename... T>
216  static auto indexBuilder(framework::pack<Cs...>, Key const&, std::tuple<T1, T...> tables)
217  {
218  static_assert(sizeof...(Cs) == sizeof...(T) + 1, "Number of columns does not coincide with number of supplied tables");
219  using tables_t = framework::pack<T...>;
220  using first_t = T1;
221  auto tail = tuple_tail(tables);
222  TableBuilder builder;
223  auto cursor = framework::FFL(builder.cursor<o2::soa::Table<Cs...>>());
224 
225  std::array<int32_t, sizeof...(T)> values;
226  iterator_tuple_t<std::decay_t<T>...> iterators = std::apply(
227  [](auto&&... x) {
228  return std::make_tuple(x.begin()...);
229  },
230  tail);
231 
232  using rest_it_t = decltype(pack_from_tuple(iterators));
233 
234  int32_t idx = -1;
235  auto setValue = [&](auto& x) -> bool {
236  using type = std::decay_t<decltype(x)>;
237  constexpr auto position = framework::has_type_at<type>(rest_it_t{});
238 
239  lowerBound<Key>(idx, x);
240  if (x == soa::RowViewSentinel{static_cast<uint64_t>(x.mMaxRow)}) {
241  return false;
242  } else if (x.template getId<Key>() != idx) {
243  return false;
244  } else {
245  values[position] = x.globalIndex();
246  ++x;
247  return true;
248  }
249  };
250 
251  auto first = std::get<first_t>(tables);
252  for (auto& row : first) {
253  idx = row.template getId<Key>();
254 
255  if (std::apply(
256  [](auto&... x) {
257  return ((x == soa::RowViewSentinel{static_cast<uint64_t>(x.mMaxRow)}) && ...);
258  },
259  iterators)) {
260  break;
261  }
262 
263  auto result = std::apply(
264  [&](auto&... x) {
265  std::array<bool, sizeof...(T)> results{setValue(x)...};
266  return (results[framework::has_type_at<std::decay_t<decltype(x)>>(rest_it_t{})] && ...);
267  },
268  iterators);
269 
270  if (result) {
271  cursor(0, row.globalIndex(), values[framework::has_type_at<T>(tables_t{})]...);
272  }
273  }
274  return builder.finalize();
275  }
276 };

293  [](auto&&... x) {
294  return std::make_tuple(x.begin()...);
295  },
296  tail);
297 
298  using rest_it_t = decltype(pack_from_tuple(iterators));
299 
300  int32_t idx = -1;
301  auto setValue = [&](auto& x) -> bool {
302  using type = std::decay_t<decltype(x)>;
303  constexpr auto position = framework::has_type_at<type>(rest_it_t{});
304 
305  lowerBound<Key>(idx, x);
306  if (x == soa::RowViewSentinel{static_cast<uint64_t>(x.mMaxRow)}) {
307  values[position] = -1;
308  return false;
309  } else if (x.template getId<Key>() != idx) {
310  values[position] = -1;
311  return false;
312  } else {
313  values[position] = x.globalIndex();
314  ++x;
315  return true;
316  }
317  };
318 
319  auto first = std::get<first_t>(tables);
320  for (auto& row : first) {
321  idx = row.template getId<Key>();
322  std::apply(
323  [&](auto&... x) {
324  (setValue(x), ...);
325  },
326  iterators);
327 
328  cursor(0, row.globalIndex(), values[framework::has_type_at<T>(tables_t{})]...);
329  }
330  return builder.finalize();
331  }

947  template <typename Task, typename R, typename C, typename Grouping, typename... Associated>
948  static void invokeProcess(Task& task, InputRecord& inputs, R (C::*)(Grouping, Associated...), std::vector<ExpressionInfo> const& infos)
949  {
950  auto tupledTask = o2::framework::to_tuple_refs(task);
951  using G = std::decay_t<Grouping>;
952  auto groupingTable = AnalysisDataProcessorBuilder::bindGroupingTable(inputs, &C::process, infos);
953 
954  // set filtered tables for partitions with grouping
955  std::apply([&groupingTable](auto&... x) {
956  (PartitionManager<std::decay_t<decltype(x)>>::setPartition(x, groupingTable), ...);
957  },
958  tupledTask);
959 
960  if constexpr (sizeof...(Associated) == 0) {
961  // single argument to process
962  std::apply([&groupingTable](auto&... x) {
963  (PartitionManager<std::decay_t<decltype(x)>>::bindExternalIndices(x, &groupingTable), ...);
964  (PartitionManager<std::decay_t<decltype(x)>>::getBoundToExternalIndices(x, groupingTable), ...);
965  },
966  tupledTask);
967  if constexpr (soa::is_soa_iterator_t<G>::value) {
968  for (auto& element : groupingTable) {
969  task.process(*element);
970  }
971  } else {
972  static_assert(soa::is_soa_table_like_t<G>::value,
973  "Single argument of process() should be a table-like or an iterator");
974  task.process(groupingTable);
975  }
976  } else {
977  // multiple arguments to process
978  static_assert(((soa::is_soa_iterator_t<std::decay_t<Associated>>::value == false) && ...),
979  "Associated arguments of process() should not be iterators");
980  auto associatedTables = AnalysisDataProcessorBuilder::bindAssociatedTables(inputs, &C::process, infos);
981  auto binder = [&](auto&& x) {
982  x.bindExternalIndices(&groupingTable, &std::get<std::decay_t<Associated>>(associatedTables)...);
983  std::apply([&x](auto&... t) {
984  (PartitionManager<std::decay_t<decltype(t)>>::setPartition(t, x), ...);
985  (PartitionManager<std::decay_t<decltype(t)>>::bindExternalIndices(t, &x), ...);
986  (PartitionManager<std::decay_t<decltype(t)>>::getBoundToExternalIndices(t, x), ...);
987  },
988  tupledTask);
989  };
990  groupingTable.bindExternalIndices(&std::get<std::decay_t<Associated>>(associatedTables)...);
991 
992  if constexpr (soa::is_soa_iterator_t<std::decay_t<G>>::value) {
993  // grouping case
994  auto slicer = GroupSlicer(groupingTable, associatedTables);
995  for (auto& slice : slicer) {
996  auto associatedSlices = slice.associatedTables();
997 
998  std::apply(
999  [&](auto&&... x) {
1000  (binder(x), ...);
1001  },
1002  associatedSlices);
1003 
1004  // bind partitions and grouping table
1005  std::apply([&groupingTable](auto&... x) {
1006  (PartitionManager<std::decay_t<decltype(x)>>::bindExternalIndices(x, &groupingTable), ...);
1007  (PartitionManager<std::decay_t<decltype(x)>>::getBoundToExternalIndices(x, groupingTable), ...);
1008  },
1009  tupledTask);
1010 
1011  invokeProcessWithArgs(task, slice.groupingElement(), associatedSlices);
1012  }
1013  } else {
1014  // non-grouping case
1015 
1016  std::apply(
1017  [&](auto&&... x) {
1018  (binder(x), ...);
1019  },
1020  associatedTables);
1021 
1022  // bind partitions and grouping table
1023  std::apply([&groupingTable](auto&... x) {
1024  (PartitionManager<std::decay_t<decltype(x)>>::bindExternalIndices(x, &groupingTable), ...);
1025  (PartitionManager<std::decay_t<decltype(x)>>::getBoundToExternalIndices(x, groupingTable), ...);
1026  },
1027  tupledTask);
1028 
1029  invokeProcessWithArgs(task, groupingTable, associatedTables);
1030  }
1031  }
1032  }
